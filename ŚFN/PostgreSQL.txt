CREATE TABLE jobs(
   job_id SERIAL,
   printer_id INTEGER,
   username TEXT NOT NULL,
   paths TEXT,
   progress TEXT,
   PRIMARY KEY(job_id),
   CONSTRAINT fk_printer
      FOREIGN KEY(printer_id)
	  REFERENCES printers(printer_id)
	  ON DELETE CASCADE
);
